// ====================== CORE ======================
public class Singleton<T> : MonoBehaviour where T : MonoBehaviour
{
private static T instance;
public static T Instance
{
get
{
if (!instance)
instance = FindObjectOfType<T>();
return instance;
}
}
protected virtual void Awake()
{
if (!instance) instance = this as T;
else Destroy(gameObject);
}
}
public static class EventBus
{
private static Dictionary<string, Action<object>> events
= new Dictionary<string, Action<object>>();
public static void Subscribe(string key, Action<object> callback)
{
if (!events.ContainsKey(key))
events[key] = callback;
else
events[key] += callback;
}
public static void Publish(string key, object data = null)
{
if (events.ContainsKey(key))
events[key]?.Invoke(data);
}
}
// ====================== STATS ======================
[Serializable]
public class CharacterStats
{
public int level = 1;
public int maxHP = 100;
public int hp;
public int attack = 10;
public int defense = 5;
public float moveSpeed = 5f;
public int exp;
public int expToNext = 100;
public void Init()
{
hp = maxHP;
}
public void GainExp(int amount)
{
exp += amount;
if (exp >= expToNext)
LevelUp();
}
void LevelUp()
{
level++;
exp = 0;
expToNext += 50;
maxHP += 30;
attack += 8;
defense += 4;
hp = maxHP;
}
}
// ====================== CHARACTER BASE ======================
public abstract class CharacterBase : MonoBehaviour
{
public CharacterStats stats;
protected Animator anim;
protected virtual void Start()
{
anim = GetComponent<Animator>();
stats.Init();
}
public virtual void TakeDamage(int dmg)
{
int final = Mathf.Max(1, dmg - stats.defense);
stats.hp -= final;
anim.SetTrigger("Hit");
if (stats.hp <= 0)
Die();
}
protected virtual void Die()
{
anim.SetTrigger("Die");
Destroy(gameObject, 3f);
}
}
// ====================== PLAYER ======================
public class PlayerController : CharacterBase
{
Rigidbody rb;
ComboSystem combo;
SkillSystem skills;
protected override void Start()
{
base.Start();
rb = GetComponent<Rigidbody>();
combo = GetComponent<ComboSystem>();
skills = GetComponent<SkillSystem>();
}
void Update()
{
Move();
AttackInput();
SkillInput();
}
void Move()
{
float h = Input.GetAxis("Horizontal");
float v = Input.GetAxis("Vertical");
Vector3 dir = new Vector3(h, 0, v);
rb.MovePosition(transform.position + dir * stats.moveSpeed *
Time.deltaTime);
anim.SetFloat("Speed", dir.magnitude);
}
void AttackInput()
{
if (Input.GetMouseButtonDown(0))
combo.Attack();
}
void SkillInput()
{
if (Input.GetKeyDown(KeyCode.Q))
skills.UseSkill(0);
if (Input.GetKeyDown(KeyCode.E))
skills.UseSkill(1);
}
}
// ====================== COMBO ======================
public class ComboSystem : MonoBehaviour
{
int comboIndex;
float lastTime;
public float resetTime = 1f;
Animator anim;
void Start()
{
anim = GetComponent<Animator>();
}
public void Attack()
{
if (Time.time - lastTime > resetTime)
comboIndex = 0;
comboIndex++;
anim.SetTrigger("Attack" + comboIndex);
if (comboIndex >= 3)
comboIndex = 0;
lastTime = Time.time;
}
}

// ====================== SKILL ======================
[CreateAssetMenu(menuName = "Skill")]
public class SkillData : ScriptableObject
{
public string skillName;
public int damage;
public float cooldown;
}
public class SkillSystem : MonoBehaviour
{
public List<SkillData> skills;
Dictionary<int, float> cooldowns = new Dictionary<int, float>();
void Update()
{
foreach (int key in new List<int>(cooldowns.Keys))
cooldowns[key] -= Time.deltaTime;
}
public void UseSkill(int index)
{
if (index >= skills.Count) return;
if (cooldowns.ContainsKey(index) && cooldowns[index] > 0) return;
SkillData skill = skills[index];
Collider[] hits = Physics.OverlapSphere(transform.position, 3f);
foreach (Collider c in hits)
{
CharacterBase cb = c.GetComponent<CharacterBase>();
if (cb && cb != GetComponent<CharacterBase>())
cb.TakeDamage(skill.damage);
}
cooldowns[index] = skill.cooldown;
}
}
// ====================== AI ======================
public class EnemyAI : CharacterBase
{
public Transform target;
NavMeshAgent agent;
protected override void Start()
{
base.Start();
agent = GetComponent<NavMeshAgent>();
agent.speed = stats.moveSpeed;
}
void Update()
{
if (!target) return;
float dist = Vector3.Distance(transform.position,
target.position);
if (dist > 2f)
{
agent.SetDestination(target.position);
anim.SetBool("Move", true);
}
else
{
anim.SetBool("Move", false);
anim.SetTrigger("Attack");
}
}
}

}
}

// ====================== STAMINA ======================
public class StaminaSystem : MonoBehaviour
{
public float maxStamina = 100f;
public float stamina;
public float regenRate = 20f;
public float regenDelay = 1f;
float lastUseTime;
void Start()
{
stamina = maxStamina;
}
void Update()
{
if (Time.time - lastUseTime > regenDelay)
{
stamina = Mathf.Min(maxStamina, stamina + regenRate *
Time.deltaTime);
}
}
public bool Use(float amount)
{
if (stamina < amount) return false;
stamina -= amount;
lastUseTime = Time.time;
return true;
}
}
// ====================== DODGE / I-FRAME ======================
public class DodgeSystem : MonoBehaviour
{
public float dodgeDistance = 5f;
public float invincibleTime = 0.5f;
bool invincible;
public void Dodge(Vector3 dir)
{
transform.position += dir.normalized * dodgeDistance;
StartCoroutine(IFrame());
}
System.Collections.IEnumerator IFrame()
{
invincible = true;
yield return new WaitForSeconds(invincibleTime);
invincible = false;
}
public bool IsInvincible()
{
return invincible;
}
}
// ====================== TARGET LOCK ======================
public class TargetLockSystem : MonoBehaviour
{
public Transform currentTarget;
public float range = 15f;
public void Lock()
{
EnemyAI[] enemies = FindObjectsOfType<EnemyAI>();
float min = Mathf.Infinity;
foreach (EnemyAI e in enemies)
{
float d = Vector3.Distance(transform.position,
e.transform.position);
if (d < min && d <= range)
{
min = d;
currentTarget = e.transform;
}
}
}
public void Unlock()
{
currentTarget = null;
}
}
// ====================== FSM BASE ======================
public interface IState
{
void Enter();
void Update();
void Exit();
}
public class StateMachine
{
IState current;
public void ChangeState(IState next)
{
current?.Exit();
current = next;
current.Enter();
}
public void Update()
{
current?.Update();
}
}

// ====================== EQUIPMENT SYSTEM ======================
public enum EquipSlot
{
Weapon,
Armor
}
[Serializable]
public class EquipmentItem : Item
{
public EquipSlot slot;
public int attackBonus;
public int defenseBonus;
}
public class EquipmentSystem : MonoBehaviour
{
public Dictionary<EquipSlot, EquipmentItem> equipped
= new Dictionary<EquipSlot, EquipmentItem>();
CharacterBase owner;
void Start()
{
owner = GetComponent<CharacterBase>();
}
public void Equip(EquipmentItem item)
{
if (equipped.ContainsKey(item.slot))
Unequip(item.slot);
equipped[item.slot] = item;
owner.stats.attack += item.attackBonus;
owner.stats.defense += item.defenseBonus;
}
public void Unequip(EquipSlot slot)
{
if (!equipped.ContainsKey(slot)) return;
EquipmentItem item = equipped[slot];
owner.stats.attack -= item.attackBonus;
owner.stats.defense -= item.defenseBonus;
equipped.Remove(slot);
}
}
