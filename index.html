<script>

// ====================== CORE ======================
class EventBus {
static events = {};
static subscribe(key, cb) {
if (!this.events[key]) this.events[key] = [];
this.events[key].push(cb);
}
static publish(key, data) {
if (!this.events[key]) return;
this.events[key].forEach(cb => cb(data));
}
}
// ====================== STATS ======================
class CharacterStats {
constructor() {
this.level = 1;
this.maxHP = 100;
this.hp = 100;
this.attack = 10;
this.defense = 5;
this.moveSpeed = 5;
this.exp = 0;
this.expToNext = 100;
}
gainExp(v) {
this.exp += v;
if (this.exp >= this.expToNext) this.levelUp();
}
levelUp() {
this.level++;
this.exp = 0;
this.expToNext += 50;
this.maxHP += 30;
this.attack += 8;
this.defense += 4;
this.hp = this.maxHP;
}
}
// ====================== CHARACTER BASE ======================
class CharacterBase {
constructor(mesh) {
this.mesh = mesh;
this.stats = new CharacterStats();
this.dead = false;
}
takeDamage(dmg) {
if (this.dead) return;
const final = Math.max(1, dmg - this.stats.defense);
this.stats.hp -= final;
if (this.stats.hp <= 0) this.die();
}
die() {
this.dead = true;
EventBus.publish("CHAR_DEAD", this);
}
update(dt) {}
}
// ====================== PLAYER ======================
class Player extends CharacterBase {
constructor(mesh) {
super(mesh);
this.combo = new ComboSystem();
this.skills = new SkillSystem(this);
this.stamina = new StaminaSystem();
}
update(dt) {
Input.update();
const dir = new THREE.Vector3(
Input.axisX,
0,
Input.axisY
);
this.mesh.position.add(dir.multiplyScalar(this.stats.moveSpeed *
dt));
if (Input.attack) this.combo.attack(this);
if (Input.skill1) this.skills.useSkill(0);
this.stamina.update(dt);
}
}
// ====================== COMBO ======================
class ComboSystem {
constructor() {
this.comboIndex = 0;
this.lastTime = 0;
}
attack(owner) {
const now = performance.now();
if (now - this.lastTime > 1000) this.comboIndex = 0;
this.comboIndex++;
if (this.comboIndex >= 3) this.comboIndex = 0;
this.lastTime = now;
}
}
// ====================== SKILL ======================
class SkillData {
constructor(name, dmg, cd) {
this.name = name;
this.damage = dmg;
this.cooldown = cd;
}
}
class SkillSystem {
constructor(owner) {
this.owner = owner;
this.skills = [
new SkillData("Shockwave", 30, 3),
new SkillData("Blast", 50, 5)
];
this.cooldowns = {};
}
update(dt) {
for (let k in this.cooldowns)
this.cooldowns[k] -= dt;
}
useSkill(i) {
if (!this.skills[i]) return;
if (this.cooldowns[i] > 0) return;
enemies.forEach(e=>{
if (e.mesh.position.distanceTo(this.owner.mesh.position) < 3)
e.takeDamage(this.skills[i].damage);
});
this.cooldowns[i] = this.skills[i].cooldown;
}
}
// ====================== STAMINA ======================
class StaminaSystem {
constructor() {
this.max = 100;
this.value = 100;
this.regen = 20;
this.delay = 1;
this.last = 0;
}
update(dt) {
if (performance.now() - this.last > this.delay*1000)
this.value = Math.min(this.max, this.value + this.regen * dt);
}
use(v) {
if (this.value < v) return false;
this.value -= v;
this.last = performance.now();
return true;
}
}
// ====================== ENEMY ======================
class Enemy extends CharacterBase {
constructor(mesh, target) {
super(mesh);
this.target = target;
}
update(dt) {
if (this.dead) return;
const dir =
this.target.mesh.position.clone().sub(this.mesh.position);
if (dir.length() > 1) {
dir.normalize();
this.mesh.position.add(dir.multiplyScalar(dt * 2));
} else {
this.target.takeDamage(this.stats.attack);
}
}
}
// ====================== BOSS ======================
class Boss extends Enemy {
die() {
super.die();
EventBus.publish("BOSS_DEAD");
}
}
// ====================== FSM ======================
class StateMachine {
change(state) {
if (this.state) this.state.exit();
this.state = state;
this.state.enter();
}
update(dt) {
if (this.state) this.state.update(dt);
}
}

class Item {

this.name = name;
this.value = value;
}
}
class Inventory {
constructor() {
this.items = [];
}
add(item) {
this.items.push(item);
}
}
// ====================== CAMERA ======================
class ThirdPersonCamera {
constructor(camera, target) {
this.camera = camera;
this.target = target;
this.offset = new THREE.Vector3(0,5,-7);
}
update() {
const pos = this.target.mesh.position.clone().add(this.offset);
this.camera.position.lerp(pos, 0.1);
this.camera.lookAt(this.target.mesh.position);
}
}

axisX:0, axisY:0,
attack:false, skill1:false,
update() {
this.attack = keys["Mouse0"];
this.skill1 = keys["KeyQ"];
}
};

window.addEventListener("mouseup",()=>keys["Mouse0"]=false);
// ====================== SCENE SETUP ======================
const scene = new THREE.Scene();
const camera = new
THREE.PerspectiveCamera(75,innerWidth/innerHeight,0.1,1000);
const renderer = new THREE.WebGLRenderer();
renderer.setSize(innerWidth,innerHeight);
document.body.appendChild(renderer.domElement);
const light = new THREE.DirectionalLight(0xffffff,1);
light.position.set(5,10,5);
scene.add(light);
const playerMesh = new THREE.Mesh(
new THREE.BoxGeometry(1,2,1),
new THREE.MeshStandardMaterial({color:0x00ff00})
);
scene.add(playerMesh);
const player = new Player(playerMesh);
const camCtrl = new ThirdPersonCamera(camera, player);
const enemies = [];
for (let i=0;i<5;i++){
const m = new THREE.Mesh(
new THREE.BoxGeometry(1,2,1),
new THREE.MeshStandardMaterial({color:0xff0000})
);
m.position.set(Math.random()*10-5,0,Math.random()*10-5);
scene.add(m);
enemies.push(new Enemy(m, player));
}
// ====================== UI ======================
function updateUI() {
document.getElementById("hp").innerText =
"HP: "+player.stats.hp+"/"+player.stats.maxHP;
document.getElementById("stamina").innerText =
"ST: "+Math.floor(player.stamina.value);
document.getElementById("exp").innerText =
"EXP: "+player.stats.exp+"/"+player.stats.expToNext;
}
// ====================== LOOP ======================
let last = performance.now();
function loop(now){
const dt = (now-last)/1000;
last = now;
player.update(dt);
enemies.forEach(e=>e.update(dt));
camCtrl.update();
updateUI();
renderer.render(scene,camera);
requestAnimationFrame(loop);
}
requestAnimationFrame(loop);
